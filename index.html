false && true || true // true
оператор && приоритетнее и сразу возвращает false, после идет сравнение false || true - оператор || возвращает true
false && (true || true); // false
по приоритетности проверяются скобки, полкчаем true, но оператор && возвращает false значение
+'40' + +'2'; // 42
так как унарный + делает из строки число
'2' + 3 ** 2; // '29'
сперва по приоритетности выполняется возведение в степень, а после соеденяется число со строкой, на выходе получаем слитую строку
3 ** 2 / 3; // 3
сперва по приоритетности выполняется возведение в степень, после - деление операндов
3 ** (9 / 3); // 27
сперва выполянется вычесление в скобках - получаем 3, после возведение в степень
'10' + (5 === 6); // '10false'
строгое равество по приоритетсности ниже чем плюс, но так как выражение в скобках сперва выполняется оно, получаем false, далее происходит конкатенация со строкой
'10' - 5 === 6; // false
строгое равество по приоритетсности ниже чем минус, поэтому сперва выполняется отнимание( 10 преобразовывется в число)- 10-5 = 5, 5 не равно 6 = false
undefined + 1; // NaN
преобразовать undefined к число невозможно и сплюсовать с числом тоже
(+null == false) < 1; // false
сперва получаем результат в скобках (null при помощи унарного плюса превращается в 0, получаем 0 == false - это true) true это 1, 1 не больше 1
(+null == false) < 1 ** 5; // false
в скобках получаем true это 1, далее по приоритетсности возведение в степень, получаем 1, 1 не равна 1
var number = 5;
2 * ++number;
2 * number--;
console.log(number); // 5
во второй строчке число увеличивается на 1 до 6 при помощи префиксного инкремента, в следующей строчке уменьшается при помощи постфиксного декремента на 1, умножение на 2 не меняет число так как это выражение не присваивается переменной
